```sql
-- 创建数据库
CREATE DATABASE IF NOT EXISTS `oj_platform` DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE `oj_platform`;

-- ----------------------------
-- 1. 用户表 user
-- ----------------------------
DROP TABLE IF EXISTS `user`;
CREATE TABLE `user` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '用户ID',
  `username` VARCHAR(50) NOT NULL COMMENT '用户名',
  `password` VARCHAR(100) NOT NULL COMMENT '密码(BCrypt加密)',
  `email` VARCHAR(100) NOT NULL COMMENT '邮箱',
  `nickname` VARCHAR(50) DEFAULT NULL COMMENT '昵称',
  `avatar` VARCHAR(255) DEFAULT NULL COMMENT '头像URL',
  `role` TINYINT NOT NULL DEFAULT 0 COMMENT '角色：0-普通用户 1-管理员',
  `phone` VARCHAR(20) DEFAULT NULL COMMENT '手机号',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  `update_time` DATETIME NOT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `UK_username` (`username`),
  UNIQUE KEY `UK_email` (`email`),
  UNIQUE KEY `UK_phone` (`phone`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- ----------------------------
-- 2. 题目分类表 category
-- ----------------------------
DROP TABLE IF EXISTS `category`;
CREATE TABLE `category` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '分类ID',
  `name` VARCHAR(50) NOT NULL COMMENT '分类名称',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  `update_time` DATETIME NOT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `UK_name` (`name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='题目分类表';

-- ----------------------------
-- 3. 题目表 problem
-- ----------------------------
DROP TABLE IF EXISTS `problem`;
CREATE TABLE `problem` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '题目ID',
  `title` VARCHAR(100) NOT NULL COMMENT '题目标题',
  `difficulty` VARCHAR(10) NOT NULL COMMENT '难度：简单/中等/困难',
  `description` TEXT NOT NULL COMMENT '题目描述',
  `input_format` TEXT NOT NULL COMMENT '输入格式',
  `output_format` TEXT NOT NULL COMMENT '输出格式',
  `sample_input` TEXT NOT NULL COMMENT '样例输入',
  `sample_output` TEXT NOT NULL COMMENT '样例输出',
  `time_limit` INT NOT NULL DEFAULT 1000 COMMENT '运行时间限制(ms)',
  `memory_limit` INT NOT NULL DEFAULT 65536 COMMENT '内存限制(KB)',
  `pass_rate` DECIMAL(5,2) DEFAULT 0.00 COMMENT '题目通过率(%)',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  `update_time` DATETIME NOT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='题目表';

-- ----------------------------
-- 4. 题目-分类关联表 problem_category
-- ----------------------------
DROP TABLE IF EXISTS `problem_category`;
CREATE TABLE `problem_category` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '关联ID',
  `problem_id` BIGINT NOT NULL COMMENT '题目ID',
  `category_id` BIGINT NOT NULL COMMENT '分类ID',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_problem_id` (`problem_id`),
  KEY `idx_category_id` (`category_id`),
  CONSTRAINT `fk_problem_category_problem` FOREIGN KEY (`problem_id`) REFERENCES `problem` (`id`),
  CONSTRAINT `fk_problem_category_category` FOREIGN KEY (`category_id`) REFERENCES `category` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='题目-分类关联表';

-- ----------------------------
-- 5. 提交记录表 submission
-- ----------------------------
DROP TABLE IF EXISTS `submission`;
CREATE TABLE `submission` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '提交ID',
  `user_id` BIGINT NOT NULL COMMENT '用户ID',
  `problem_id` BIGINT NOT NULL COMMENT '题目ID',
  `problem_title` VARCHAR(100) NOT NULL COMMENT '题目标题',
  `language` VARCHAR(20) NOT NULL COMMENT '语言：C++/Java',
  `code` LONGTEXT NOT NULL COMMENT '提交代码',
  `status` TINYINT NOT NULL DEFAULT 0 COMMENT '评测状态',
  `pass_count` INT DEFAULT NULL COMMENT '通过用例数',
  `total_count` INT DEFAULT NULL COMMENT '总用例数',
  `run_time` INT DEFAULT NULL COMMENT '运行时间(ms)',
  `memory` INT DEFAULT NULL COMMENT '内存占用(KB)',
  `error_msg` TEXT DEFAULT NULL COMMENT '错误信息',
  `submit_time` DATETIME NOT NULL COMMENT '提交时间',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_problem_id` (`problem_id`),
  CONSTRAINT `fk_submission_user` FOREIGN KEY (`user_id`) REFERENCES `user` (`id`),
  CONSTRAINT `fk_submission_problem` FOREIGN KEY (`problem_id`) REFERENCES `problem` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='提交记录表';

-- ----------------------------
-- 6. AI评价表 evaluation
-- ----------------------------
DROP TABLE IF EXISTS `evaluation`;
CREATE TABLE `evaluation` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '评价ID',
  `submission_id` BIGINT NOT NULL COMMENT '提交ID',
  `time_complexity` VARCHAR(50) DEFAULT NULL COMMENT '时间复杂度',
  `space_complexity` VARCHAR(50) DEFAULT NULL COMMENT '空间复杂度',
  `code_style` TEXT DEFAULT NULL COMMENT '代码风格评价',
  `readability` TEXT DEFAULT NULL COMMENT '可读性评价',
  `optimization` TEXT DEFAULT NULL COMMENT '优化建议',
  `ai_model` VARCHAR(50) DEFAULT NULL COMMENT '调用AI模型',
  `cost` DECIMAL(10,4) DEFAULT NULL COMMENT '调用成本',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_submission_id` (`submission_id`),
  CONSTRAINT `fk_evaluation_submission` FOREIGN KEY (`submission_id`) REFERENCES `submission` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='AI评价表';

-- ----------------------------
-- 7. 题解表 solution
-- ----------------------------
DROP TABLE IF EXISTS `solution`;
CREATE TABLE `solution` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '题解ID',
  `problem_id` BIGINT NOT NULL COMMENT '题目ID',
  `title` VARCHAR(100) NOT NULL COMMENT '题解标题',
  `content` LONGTEXT NOT NULL COMMENT '题解内容',
  `language` VARCHAR(20) NOT NULL COMMENT '题解语言',
  `code` LONGTEXT NOT NULL COMMENT '题解代码',
  `create_user_id` BIGINT DEFAULT NULL COMMENT '创建人',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  `update_time` DATETIME NOT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_problem_id` (`problem_id`),
  CONSTRAINT `fk_solution_problem` FOREIGN KEY (`problem_id`) REFERENCES `problem` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='题解表';

-- ----------------------------
-- 8. 用户Token表 user_token
-- ----------------------------
DROP TABLE IF EXISTS `user_token`;
CREATE TABLE `user_token` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `user_id` BIGINT NOT NULL COMMENT '用户ID',
  `token` VARCHAR(500) NOT NULL COMMENT '访问令牌(JWT)',
  `expire_time` DATETIME NOT NULL COMMENT '过期时间',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  `update_time` DATETIME NOT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_token` (`token`),
  KEY `idx_user_id` (`user_id`),
  CONSTRAINT `fk_user_token_user` FOREIGN KEY (`user_id`) REFERENCES `user` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户Token表';

-- ----------------------------
-- 9. 用户做题统计表 user_statistics
-- ----------------------------
DROP TABLE IF EXISTS `user_statistics`;
CREATE TABLE `user_statistics` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `total_submit` INT DEFAULT 0 COMMENT '总提交次数',
  `total_accept` INT DEFAULT 0 COMMENT '总通过次数',
  `create_time` DATETIME NOT NULL,
  `update_time` DATETIME NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `UK_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 10. 用户题目状态表 user_problem
-- ----------------------------
DROP TABLE IF EXISTS `user_problem`;
CREATE TABLE `user_problem` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `user_id` BIGINT NOT NULL COMMENT '用户ID',
  `problem_id` BIGINT NOT NULL COMMENT '题目ID',
  
  -- 状态：0=未通过/尝试过  1=已通过
  `status` TINYINT NOT NULL DEFAULT 0 COMMENT '题目状态：0-未通过 1-已通过',
  
  `last_submit_time` DATETIME DEFAULT NULL COMMENT '最后提交时间',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  `update_time` DATETIME NOT NULL COMMENT '更新时间',

  PRIMARY KEY (`id`),
  -- 联合唯一索引：一个用户对一道题只能有一条记录
  UNIQUE KEY `uk_user_problem` (`user_id`,`problem_id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_problem_id` (`problem_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户题目状态表';

-- ----------------------------
-- 11. 题目测试样例表 problem_test_case
-- ----------------------------
--
DROP TABLE IF EXISTS `problem_test_case`;
CREATE TABLE `problem_test_case` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '测试样例ID',
  `problem_id` BIGINT NOT NULL COMMENT '题目ID',
  
  `input_url` VARCHAR(512) NOT NULL COMMENT '输入文件URL',
  `output_url` VARCHAR(512) NOT NULL COMMENT '输出文件URL',
  
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  `update_time` DATETIME NOT NULL COMMENT '更新时间',

  PRIMARY KEY (`id`),
  KEY `idx_problem_id` (`problem_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='题目测试样例表';
```
